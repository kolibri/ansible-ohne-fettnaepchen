```yaml title="The ini module"
- name: ensure git config
  ini_file:
    dest: "{{ ansible_user_dir }}/.gitconfig"
    section: "{{ item.ns }}"
    option: "{{ item.name }}"
    value: "{{ item.value }}"
  loop:
    - { ns: user, name: email,   value: "{{ git_user_mail }}" }
    - { ns: user, name: name,    value: "{{ git_user_name }}" }
    - { ns: push, name: default, value: simple }
```
